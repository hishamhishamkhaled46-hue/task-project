<?php
// session_start();
include("inc/sit_header.php");

// --- منطق الفلترة والترقيم الصفحي ---
$limit = 12; // تم زيادة العدد ليكون العرض أفضل
$page = isset($_GET['page']) ? (int)$_GET['page'] : 1;
if ($page < 1) $page = 1;
$start = ($page - 1) * $limit;

// استقبال نوع الترتيب (الفلترة)
$sort = isset($_GET['sort']) ? $_GET['sort'] : 'default';
$order_by = "ORDER BY id DESC"; // الافتراضي
$where_clause = "";

// تحديد الفلترة بناءً على القسم أو الماركة
if (isset($_GET['catid'])) {
    $cat_id = $_GET['catid'];
    $where_clause = "WHERE cat = '$cat_id'";
} elseif (isset($_GET['braid'])) {
    $bra_id = $_GET['braid'];
    $where_clause = "WHERE brand = '$bra_id'";
}

// إضافة منطق الترتيب (Sorting)
switch ($sort) {
    case 'popular':
        $order_by = "ORDER BY views DESC"; // ترتيب حسب الأكثر مشاهدة
        break;
    case 'low_price':
        $order_by = "ORDER BY price ASC"; // ترتيب حسب الأرخص
        break;
    case 'offers':
        $where_clause .= (empty($where_clause) ? "WHERE" : " AND") . " discount > 0";
        $order_by = "ORDER BY discount DESC"; // ترتيب حسب أعلى خصم
        break;
}

// استعلام العد لجلب إجمالي المنتجات بعد الفلترة
$count_query = "SELECT COUNT(id) AS total FROM products $where_clause";
$count_result = $conn->query($count_query);
$total_products = $count_result->fetch_assoc()['total'];
$total_pages = ceil($total_products / $limit);

// استعلام جلب البيانات النهائي
$select = "SELECT * FROM products $where_clause $order_by LIMIT $start, $limit";
$result = $conn->query($select);
?>

                        <ol class="breadcrumb justify-content-lg-end mb-0 px-0">

            </div>
        </div>

    <div class="dropdown my-4">

            <i class="fas fa-sort-amount-down mr-2"></i> Filtering
        </button>
        <ul class="dropdown-menu shadow" aria-labelledby="filterDropdown">
            <li><a class="dropdown-item" href="?sort=popular<?=isset($cat_id)?'&catid='.$cat_id:''?><?=isset($bra_id)?'&braid='.$bra_id:''?>">Most Popular</a></li>
            <li><a class="dropdown-item" href="?sort=low_price<?=isset($cat_id)?'&catid='.$cat_id:''?><?=isset($bra_id)?'&braid='.$bra_id:''?>">Lowest Price</a></li>
            <li><hr class="dropdown-divider"></li>
            <li><a class="dropdown-item text-danger" href="?sort=offers<?=isset($cat_id)?'&catid='.$cat_id:''?><?=isset($bra_id)?'&braid='.$bra_id:''?>">Special Offers</a></li>
        </ul>
    </div>

    <section class="py-5">
        <div class="container p-0">
            <div class="row">
                <?php include("inc/sidebar.php"); ?>

                <div class="col-lg-9 order-1 order-lg-2 mb-5 mb-lg-0">
                    <div class="row mb-3 align-items-center">
                        <div class="col-lg-6 mb-2 mb-lg-0">
                            <p class="text-small text-muted mb-0">Showing <?php echo ($total_products > 0) ? $start + 1 : 0; ?>–<?php echo min($start + $limit, $total_products); ?> of <?php echo $total_products; ?> results</p>
                        </div>
                    </div>

                    <div class="row">
                        <?php 
                        if ($result->num_rows > 0):
                            while ($row = $result->fetch_assoc()): 
                                $images_array = explode(',', $row['image']); 
                        ?>
                                <div class="col-xl-4 col-lg-4 col-sm-6 mb-4">
                                    <div class="product text-center">
                                        <div class="position-relative mb-3">
                                            <?php if(isset($row['discount']) && $row['discount'] > 0): ?>
                                                <div class="badge text-white badge-danger">Sale</div>
                                            <?php else: ?>
                                                <div class="badge text-white badge-info">New</div>
                                            <?php endif; ?>
                                            
                                            <a class="d-block" href="detail.php?id=<?=$row['id']?>">
                                                <img class="img-fluid w-100" src="admin/image/<?=$images_array[0]?>" alt="<?=$row['name']?>" style="height:250px; object-fit: cover;">
                                            </a>
                                            
                                            <div class="product-overlay">
                                                <ul class="mb-0 list-inline">
                                                    <li class="list-inline-item m-0 p-0">
                                                        <a class="btn btn-sm btn-outline-dark" href="#"><i class="far fa-heart"></i></a>
                                                    </li>
                                                    <li class="list-inline-item m-0 p-0">
                                                        <a class="btn btn-sm btn-dark" href="ADD_CART.php?proid=<?=$row['id']?>">Add to cart</a>
                                                    </li>
                                                    <li class="list-inline-item mr-0">
                                                        <a class="btn btn-sm btn-outline-dark" href="detail.php?id=<?=$row['id']?>">
                                                            <i class="fas fa-expand"></i>
                                                        </a>
                                                    </li>
                                                </ul>
                                            </div>
                                        </div>
                                        <h6> <a class="reset-anchor" href="detail.php?id=<?=$row['id']?>"><?=$row['name']?></a></h6>
                                        <p class="small text-muted"><?=$row['price']?> EGP</p>
                                    </div>
                                </div>
                        <?php 
                            endwhile; 
                        else:
                            echo "<div class='col-12'><p class='alert alert-info text-center'>No products found.</p></div>";
                        endif;
                        ?>
                    </div>

                    <nav aria-label="Page navigation example" class="mt-4">
                        <ul class="pagination justify-content-center justify-content-lg-end">
                            <li class="page-item <?php if($page <= 1) echo 'disabled'; ?>">
                                <a class="page-link" href="?page=<?php echo $page - 1; ?>&sort=<?=$sort?><?php echo isset($cat_id) ? '&catid='.$cat_id : ''; ?><?php echo isset($bra_id) ? '&braid='.$bra_id : ''; ?>">&laquo;</a>
                            </li>

                            <?php for ($i = 1; $i <= $total_pages; $i++): ?>
                                <li class="page-item <?php if($page == $i) echo 'active'; ?>">
                                    <a class="page-link" href="?page=<?php echo $i; ?>&sort=<?=$sort?><?php echo isset($cat_id) ? '&catid='.$cat_id : ''; ?><?php echo isset($bra_id) ? '&braid='.$bra_id : ''; ?>"><?php echo $i; ?></a>
                                </li>
                            <?php endfor; ?>

                            <li class="page-item <?php if($page >= $total_pages) echo 'disabled'; ?>">
                                <a class="page-link" href="?page=<?php echo $page + 1; ?>&sort=<?=$sort?><?php echo isset($cat_id) ? '&catid='.$cat_id : ''; ?><?php echo isset($bra_id) ? '&braid='.$bra_id : ''; ?>">&raquo;</a>
                            </li>
                        </ul>
                    </nav>
                </div>
            </div>
        </div>
    </section>
</div>
